#=====================Project-1(BMI Calculator)==========================

w = float(input("Enter the your Perfect Weight(kg): "))
h = float(input("Enter the your Perfect Height(m): "))

t = ((w)/(h**2))

print("Your Body Mass Index(BMI): ",t)

if t<18.5:
    print("You are Underweight...")
elif 18.5<= t <24.9:
    print("You have a normal Weight...")
elif 25<= t <29.9:
    print("You are overweight...")
else:
    print("You are obese...")
